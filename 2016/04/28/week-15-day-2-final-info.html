Assignment 3 # 2 Solution

(define (cache f)
    (define locals (get 'parameters f))
    (define denv (get' __context f))
    (define (replace code)
        (cond
            ((null? code) nil)
            ((and (symbol? code) (member? code locals))
                code
                )
            ((symbol? code)
                (get code denv)
                )
            ((and (pair? code) (eq? (car code) 'define) (pair? (cadr code)))
                (set! locals (cons (car (cadr code)) locals))
                code
                )
            ((and (pair? code) (eq? (car code) 'define))
                (set! locals (cons (cadr code) locals))
                (cons 'define (cons (car code) (replace (cddr code))))
                )
            ((pair? code)
                (cons (replace (car code)) (replace (cdr code)))
                )
            (else
                code
                )
            )
        )
    (set 'code (replace (get 'code f)) f)
)
